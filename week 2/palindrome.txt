import java.io.*;
import java.util.*;
class UserMainCode
{
public int isPalindrome(String input1)
{
 String str = input1.toLowerCase();
 int left = 0;
 int right = str.length() - 1;
 while (left < right)
 {
 if (str.charAt(left) != str.charAt(right))
 {
 return 1; 
 }
 left++;
 right--;
 }
 return 2;
}
